typedef struct {
    double min, max, avg;
    int count;
} Stats;

Stats calc_stats(const char *fname) {
    Stats s = {DBL_MAX, -DBL_MAX, 0, 0};
    FILE *fp = fopen(fname, "r");
    double val, sum = 0;
    
    while (fscanf(fp, "%lf", &val) == 1) {
        if (val < s.min) s.min = val;
        if (val > s.max) s.max = val;
        sum += val;
        s.count++;
    }
    
    s.avg = (s.count > 0) ? 
            sum / s.count : 0;
    fclose(fp);
    return s;
}
